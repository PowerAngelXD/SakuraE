# Pre Define
<number> -> Numbers includes Integer and Decimal
<string> -> Text like: "content"
<char> -> Single character: 'char'
<field> -> Raw identifier.

# Define
Literal -> <number> | <string> | 'true' | 'false' | <char>
IndexOp -> '[' AddExpr ']'
CallingOp -> '(' WholeExpr (',' WholeExpr)* ')'
AtomIdentifierExpr -> Identifier (IndexOp|CallingOp)*
IdentifierExpr -> AtomIdentifierExpr ('.' AtomIdentifierExpr)*
PrimExpr -> Literal | IdentifierExpr

MulExpr -> PrimExpr (('*'|'/') PrimExpr)*
AddExpr -> MulExpr (('+'|'-') MulExpr)*
LogicExpr -> AddExpr (('=='|'!='|'>'|'<'|'<='|'>=') AddExpr)*
BinaryExpr -> LogicExpr (('||'|'&&') LogicExpr)*
ArrayExpr -> '[' WholeExpr （',' WholeExpr）* ']'
AssignExpr -> IdentifierExpr '=' WholeExpr

WholeExpr -> AddExpr | BinaryExpr | ArrayExpr | AssignExpr

BasicTypeModifier -> 'int'|'char'|'raw_string'|'float'|'double'|＜field>
ArrayTypeModifier ->  '[' <number> ']' BasicTypeModifier
TypeModifier -> ArrayTypeModifier | BasicTypeModifier

DeclareStmt -> 'let' <field> (':' TypeModifier) '=' WholeExpr '；'
ExprStmt -> IdentifierExpr | AssignExpr ';'
BlockStmt -> '{' (Stmt)* '}'
IfStmt -> 'if' '(' BinaryExpr ')' Stmt | BlockStmt
ElseStmt -> 'else' Stmt | BlockStmt
WhileStmt -> 'while' '(' BinaryExpr ')' Stmt | BlockStmt
ForStmt -> 'for' '(' LetStmt BinaryExpr ';' WholeExpr ';' ')' Stmt | BlockStmt
FuncDefineStmt -> 'func' <field> '(' (<field> ':' TypeModifier) (',' <field> ':' TypeModifier) ')' '->' TypeModifier BlockStmt
ReturnStmt -> 'return' WholeExpr ';'
Stmt -> DeclareStmt | ExprStmt | BlockStmt | IfStmt | ElseStmt | WhileStmt | ForStmt | ReturnStmt
